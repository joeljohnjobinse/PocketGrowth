"use client";

import { useState } from "react";
import { supabase } from "@/lib/supabaseClient";

interface AddAllowanceFormProps {
  userId: string;
  onUpdate: (newLockedAmount: number) => void;
}

export default function AddAllowanceForm({ userId, onUpdate }: AddAllowanceFormProps) {
  const [amount, setAmount] = useState<number>(0);
  const [message, setMessage] = useState<string>("");

  const handleAddAllowance = async () => {
    try {
      // Insert allowance transaction
      const { error: txnError } = await supabase.from("transactions").insert({
        user_id: userId,
        amount,
        type: "allowance",
      });

      if (txnError) throw txnError;

      setMessage(`$${amount.toFixed(2)} has been added to your balance.`);

      setAmount(0); // reset input

      // Fetch updated locked amount
      const { data: savingsData, error: savingsError } = await supabase
        .from("savings")
        .select("locked_amount")
        .eq("user_id", userId)
        .single();

      if (savingsError && savingsError.code !== "PGRST116") throw savingsError;

      const newLocked = savingsData ? Number(savingsData.locked_amount) : 0;
      onUpdate(newLocked);
    } catch (err) {
      console.error("Error in handleAddAllowance:", err);
      setMessage("Failed to add allowance. Check console.");
    }
  };

  return (
    <div>
      <h3 className="text-lg font-semibold mb-4">Add Allowance</h3>
      <input
        type="number"
        className="border p-2 rounded w-full mb-4"
        value={amount || ""}
        onChange={(e) => setAmount(parseFloat(e.target.value))}
        placeholder="Enter amount"
      />
      <button
        onClick={handleAddAllowance}
        className="bg-blue-600 text-white py-2 px-4 rounded"
      >
        Add Allowance
      </button>
      {message && <p className="mt-2 text-green-600">{message}</p>}
    </div>
  );
}
